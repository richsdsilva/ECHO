    class Particle {
    constructor() { this.reset(); }
    reset() {
    this.x = Math.random() * 200;
    this.y = Math.random() * 200;
    this.size = Math.random() * 1.0 + 0.2;
    this.alpha = Math.random() * 0.28 + 0.07;
    this.vx = (Math.random() - 0.5) * 0.3;
    this.vy = (Math.random() - 0.5) * 0.3;
    }
    update() {
    let dx = CX - this.x;
    let dy = CY - this.y;
    let dist = Math.sqrt(dx*dx + dy*dy) || 1;
    if (suctionStrength > 0) {
    this.x += (dx / dist) * (suctionStrength * 0.5);
    this.y += (dy / dist) * (suctionStrength * 0.5);
    }
    this.x += this.vx;
    this.y += this.vy;
    this.alpha -= 0.0008;
    if (this.alpha <= 0 || dist < 5) this.reset();
    }

    cx.fillStyle = `rgba(255, 255, 255, ${Math.max(0, this.alpha)})`;
    cx.beginPath();
    cx.arc(this.x, this.y, this.size, 0, Math.PI*2);

    if(!parsed.verdict || !parsed.sub) throw new Error('Campos ausentes');
    renderVerdict(parsed.verdict, parsed.sub, parsed.tag || 'echo', input);
    } catch(err) {
    // Mostra erro real para facilitar debug
    renderVerdict('ERRO', err.message.slice(0,80), 'echo', input);
    }
    }
